00144 int create_simptcp_socket()
00145 {
00146   int fd;
00147 
00148   /* get a free simptcp socket descriptor */
00149   for (fd=0;fd< MAX_OPEN_SOCK;fd++) {
00150     if ((simptcp_entity.simptcp_socket_descriptors[fd]) == NULL){ 
00151       /* this is a free descriptor */
00152       /* Allocating memory for the new simptcp_socket */
00153       simptcp_entity.simptcp_socket_descriptors[fd] = 
00154         (struct simptcp_socket *) malloc(sizeof(struct simptcp_socket));
00155       if (!simptcp_entity.simptcp_socket_descriptors[fd]) {
00156         return -ENOMEM;
00157       }
00158       /* initialize the simptcp socket control block with
00159          local port number set to 15000+fd */
00160       init_simptcp_socket(simptcp_entity.simptcp_socket_descriptors[fd],15000+fd);
00161       simptcp_entity.open_simptcp_sockets++;
00162       
00163       /* return the socket descriptor */
00164       return fd;
00165     }
00166   } /* for */
00167   /* The maximum number of open simptcp
00168      socket reached  */
00169   return -ENFILE;
